... with configurable Ranges of selectable Dates

Default Design

White Design

Show Example Code
var month = new Date().getMonth();

tail.DateTime("#datetime-3", {
    dateRanges: [
        [new Date(2018, month, 12), new Date(2018, month, 26)],
        ["2018-" + month.toString() + "-4", "2018-" + month.toString() + "-30"],
        [new Date(2018, month+1, 1), new Date(2018, month+1, 31)],
        ["MON", "FRI"]
    ],

    static: "#datetime-3-holder",   /* Used for demonstration */
    classNames: "theme-default",    /* Used for demonstration */
    startOpen: true,                /* Used for demonstration */
    stayOpen: true,                 /* Used for demonstration */
});
